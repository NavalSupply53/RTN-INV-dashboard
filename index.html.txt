<script>
    // Configuration
    const COORDS_SATTAHIP = [12.6569, 100.8842];
    const COORDS_SONGKHLA = [7.2023, 100.5750];
    // ใช้ Proxy เพื่อเลี่ยงปัญหา CORS
    const PROXY_URL = "https://api.allorigins.win/raw?url=";
    const SHEET_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vTqKTsZLO8hEYr4eSK1imLpVUDAaP-UGCrqTYcHMPdRnXZwit1GmpKCPOdMwC6t5QyVVcKWgm7JFIq8/pub?output=csv";
    
    const GID_BASE = "0";
    const GID_SHIP = "421033282";
    const GID_MEDICAL = "1121118182";

    // Initialize Map
    const map = L.map('map').setView([10.0, 101.0], 6);
    L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png').addTo(map);

    L.marker(COORDS_SATTAHIP).addTo(map).bindPopup("ฐานทัพเรือสัตหีบ");
    L.marker(COORDS_SONGKHLA).addTo(map).bindPopup("ฐานทัพเรือสงขลา");

    const lineSat = L.polyline([], {color: '#ff4444', dashArray: '5, 10', weight: 1}).addTo(map);
    const lineSon = L.polyline([], {color: '#4444ff', dashArray: '5, 10', weight: 1}).addTo(map);

    map.on('mousemove', (e) => {
        const cursor = e.latlng;
        lineSat.setLatLngs([COORDS_SATTAHIP, cursor]);
        lineSon.setLatLngs([COORDS_SONGKHLA, cursor]);

        const dSatNM = (turf.distance(turf.point([COORDS_SATTAHIP[1], COORDS_SATTAHIP[0]]), turf.point([cursor.lng, cursor.lat])) / 1.852).toFixed(2);
        const dSonNM = (turf.distance(turf.point([COORDS_SONGKHLA[1], COORDS_SONGKHLA[0]]), turf.point([cursor.lng, cursor.lat])) / 1.852).toFixed(2);

        $('#cursor-tooltip').show().css({ left: e.containerPoint.x + 15, top: e.containerPoint.y + 15 });
        $('#dist-sattahip').html(`Sattahip: ${dSatNM} NM`);
        $('#dist-songkhla').html(`Songkhla: ${dSonNM} NM`);
    });

    const drawnItems = new L.FeatureGroup().addTo(map);
    new L.Control.Draw({
        draw: { polyline: false, polygon: true, rectangle: true, circle: true, marker: true },
        edit: { featureGroup: drawnItems }
    }).addTo(map);
    map.on(L.Draw.Event.CREATED, (e) => drawnItems.addLayer(e.layer));

    // Data Fetching with exponential backoff and proxy
    async function fetchSheetData(gid) {
        const url = `${PROXY_URL}${encodeURIComponent(SHEET_URL + '&gid=' + gid)}`;
        let retries = 5;
        let delay = 1000;

        while (retries > 0) {
            try {
                const response = await fetch(url);
                if (!response.ok) throw new Error('Network response was not ok');
                const csvText = await response.text();
                
                return new Promise((resolve) => {
                    Papa.parse(csvText, {
                        header: true,
                        skipEmptyLines: true,
                        complete: (results) => resolve(results.data)
                    });
                });
            } catch (err) {
                retries--;
                if (retries === 0) throw err;
                await new Promise(r => setTimeout(r, delay));
                delay *= 2;
            }
        }
    }

    let chartSat, chartSon;
    function updateBaseCharts(data) {
        if (!data || data.length === 0) return;
        const satData = data.find(d => d.Base_Name === "Sattahip") || {MRE:0, Fuel:0, Harpoon:0, HarpoonB1C:0, ESSM:0};
        const sonData = data.find(d => d.Base_Name === "Songkhla") || {MRE:0, Fuel:0, Harpoon:0, HarpoonB1C:0, ESSM:0};
        const labels = ['MRE', 'Fuel', 'Harpoon', 'HarpoonB1C', 'ESSM'];
        const colors = ['#28a745', '#ffc107', '#dc3545', '#17a2b8', '#fd7e14'];

        const config = (vals) => ({
            type: 'bar',
            data: { labels, datasets: [{ data: vals, backgroundColor: colors }] },
            options: { indexAxis: 'y', plugins: { legend: { display: false } }, scales: { x: { beginAtZero: true, grid: { color: '#333' } } }, color: '#fff' }
        });

        if(chartSat) chartSat.destroy();
        chartSat = new Chart(document.getElementById('chartSattahip'), config([satData.MRE, satData.Fuel, satData.Harpoon, satData.HarpoonB1C, satData.ESSM]));
        
        if(chartSon) chartSon.destroy();
        chartSon = new Chart(document.getElementById('chartSongkhla'), config([sonData.MRE, sonData.Fuel, sonData.Harpoon, sonData.HarpoonB1C, sonData.ESSM]));
    }

    function updateShipTable(data) {
        const tbody = $('#ship-table-body').empty();
        if (!data) return;
        data.forEach((ship, i) => {
            const score = parseInt(ship.Readiness_Score) || 0;
            const color = score > 80 ? 'text-success' : score >= 50 ? 'text-warning' : 'text-danger';
            tbody.append(`
                <tr>
                    <td>${i+1}</td>
                    <td>${ship.Ship_Name || 'N/A'}</td>
                    <td class="text-center fw-bold ${color}">${score}%</td>
                    <td><span class="badge ${score > 80 ? 'bg-success' : score >= 50 ? 'bg-warning text-dark' : 'bg-danger'}">${ship.Status || '-'}</span></td>
                    <td><small>${ship.Remark || ''}</small></td>
                </tr>
            `);
        });
    }

    function updateMedicalCharts(data) {
        const container = $('#hospital-container').empty();
        if (!data) return;
        data.forEach((h, i) => {
            const id = `hosp-${i}`;
            container.append(`<div class="mb-4"><canvas id="${id}"></canvas></div>`);
            const total = parseInt(h.Total_Beds) || 1;
            const occupied = parseInt(h.Occupied_Beds) || 0;
            
            new Chart(document.getElementById(id), {
                type: 'doughnut',
                data: {
                    labels: ['ใช้ไป', 'ว่าง'],
                    datasets: [{ data: [occupied, Math.max(0, total - occupied)], backgroundColor: ['#dc3545', '#28a745'] }]
                },
                options: { 
                    plugins: { 
                        title: { display: true, text: h.Hospital_Name || 'Unknown', color: '#fff' },
                        legend: { position: 'bottom', labels: { color: '#ccc', font: { size: 10 } } }
                    } 
                }
            });
        });
    }

    async function refreshDashboard() {
        $('#loader').show();
        $('#error-status').text('');
        try {
            const [baseData, shipData, medData] = await Promise.all([
                fetchSheetData(GID_BASE),
                fetchSheetData(GID_SHIP),
                fetchSheetData(GID_MEDICAL)
            ]);

            updateBaseCharts(baseData);
            updateShipTable(shipData);
            updateMedicalCharts(medData);
            
            $('#last-update').text(`อัปเดตล่าสุด: ${new Date().toLocaleTimeString()}`);
            $('#loader').fadeOut();
        } catch (error) {
            console.error("Error loading data:", error);
            $('#error-status').text('เกิดข้อผิดพลาดในการเชื่อมต่อข้อมูล กรุณาตรวจสอบการเผยแพร่ไฟล์ Google Sheets');
            // หลังจาก retry แล้วยังพัง ให้ซ่อน loader เพื่อให้ใช้งานส่วนอื่นๆ ได้
            setTimeout(() => $('#loader').fadeOut(), 3000);
        }
    }

    $(document).ready(() => {
        refreshDashboard();
        setInterval(refreshDashboard, 300000); // อัปเดตทุก 5 นาที
    });
</script>
