                Repair_Time: ''
            };
            ships.unshift(newShip);
            saveData();
            renderShips();
        }

        function deleteShip(id) {
            if(confirm("ยืนยันการลบข้อมูลเรือลำนี้?")) {
                ships = ships.filter(s => s.id !== id);
                saveData();
                renderShips();
            }
        }

        function updateShip(index, field, value) {
            if(['name', 'Repair_Location', 'Repair_Time'].includes(field)) {
                ships[index][field] = value;
            } else {
                ships[index][field] = parseFloat(value) || 0;
            }

            // คำนวณ Current อัตโนมัติเมื่อมีการอัปเดตค่าที่เกี่ยวข้อง
            const s = ships[index];

            // 1. คำนวณ Food Current = Max - (Rate * Days)
            if (['Food_Max', 'Food_Rate', 'Food_Days'].includes(field)) {
                let calc = s.Food_Max - ((s.Food_Rate || 0) * (s.Food_Days || 0));
                s.Food_Current = Math.max(0, calc).toFixed(1); // ไม่ให้ค่าติดลบ
            }

            // 2. คำนวณ Water Current = Max - (Rate * Days)
            if (['Water_Max', 'Water_Rate', 'Water_Days'].includes(field)) {
                let calc = s.Water_Max - ((s.Water_Rate || 0) * (s.Water_Days || 0));
                s.Water_Current = Math.max(0, calc).toFixed(1);
            }

            // 3. คำนวณ Fuel Current จาก %
            if (['Fuel_Max', 'Fuel_Percent'].includes(field)) {
                if(s.Fuel_Percent > 100) s.Fuel_Percent = 100;
                if(s.Fuel_Percent < 0) s.Fuel_Percent = 0;
                s.Fuel_Current = (s.Fuel_Max * (s.Fuel_Percent / 100)).toFixed(0);
            }

            saveData();
            renderShips(false);
        }

        function getShipNameStatusClass(hp) {
            if (hp <= 0) return 'status-inactive'; 
            if (hp <= 30) return 'status-critical'; 
            if (hp <= 79) return 'status-warning'; 
            return 'status-normal'; 
        }

        function saveBaseData() {
            const baseId = currentEditingBase;
            const inputs = document.querySelectorAll('#modal-inputs input');
            inputs.forEach(input => {
                const key = input.id.replace('input-', '');
                baseData[baseId][key] = parseFloat(input.value) || 0;
            });
            const keys = ['MRE', 'Fuel', 'Harpoon', 'HarpoonB1C', 'ESSM'];
            charts[baseId].data.datasets[0].data = keys.map(k => (baseData[baseId][`${k}_Current`] / (baseData[baseId][`${k}_Max`] || 1) * 100));
            charts[baseId].update();
            saveData();
            closeModal();
        }

            saveData();
            renderHospitals();
        }
